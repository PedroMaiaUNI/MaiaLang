 * Programa de Teste Simples para o Scanner/Parser
 */

// declaração básica
int contador;
float valor_pi = 3.14159;

// função
void calcular(int limite, float taxa) {
    contador = 0; // atribuição
    int i; // variável local
    
    // while-loop
    while (i <= limite) {
        i = i + 1;
        contador = contador + 1;
    }
    
    // condicional
    if (taxa > 10.0) {
        // chamada de função
        imprimir_resultado(contador); 
    } else {
        valor_pi = valor_pi * taxa;
    }
}

// função com retorno
int imprimir_resultado(int n) {
    if (n != 0) {
        return n * 2;
    }
    return 0;
}
